class Solution {
public:
    int subarraysDivByK(vector<int>& nums, int k) {
        
        int prefix = 0;

        int i = 0, j = 0, n = nums.size(), ans = 0;

        map <int, int> mp;
        mp[0] = 1;

        while (i < n) {
            prefix += nums[i];
            int prefixMod = ((prefix % k) + k) % k;

            if (mp.find(prefixMod) != mp.end()) {
                ans += mp[prefixMod];
            }

            mp[prefixMod]++;
            i++;
        }

        return ans;
    }
};
